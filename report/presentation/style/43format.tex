\newcommand{\thankyou}[3]{
	\setbeamercolor{normal text}{fg=white}\usebeamercolor*{normal text}
	\begin{frame}[plain]
		\begin{tikzpicture}[overlay, remember picture]
		\clip (current page.north west) rectangle (current page.south east);
		\shade[%
		top color=CD03,
		bottom color=CD02,
		shading angle=-45,
		%opacity=.8
		]
		(current page.north west) rectangle (current page.south east);
		\node[anchor=north west] at ([xshift=1cm, yshift=-0.5cm]current page.north west) {\unilogow{5cm}};
		\if@guestlogo
		\node[anchor=north west] at ([xshift=6.5cm, yshift=-0.5cm]current page.north west) {\guestlogo{5cm}};
		\fi
		\if@department
		\node[anchor=north east] at ([xshift=11.8cm, yshift=-0.5cm]current page.north west) {\pgfuseimage{department}};
		\fi
		\if@depunilogo
		\node[anchor=north east] at ([xshift=11.8cm, yshift=-0.5cm]current page.north west) {\pgfuseimage{department}};
		\fi
		\node[anchor=north west] at ([xshift=4.4cm, yshift=-2.5cm]current page.north west) {
			\if@german
			\Huge\textbf{\textcolor{white}{Vielen Dank!}}
			\else	
			\Huge\textbf{\textcolor{white}{Thank You!}}
			\fi
		};
		\end{tikzpicture}
		
		\vspace{4cm}
		
		\begin{columns}[c]
			\begin{column}{2.5cm}
				
				\begin{tikzpicture}
				\node[circle,minimum size=2.5cm, path picture={
					\node at (path picture bounding box.center){\pgfuseimage{speaker}};}] {};
				\end{tikzpicture}
			\end{column}
			\begin{column}{.7\textwidth}
				
				\begin{beamercolorbox}%[leftskip=3ex, ht=2em,wd=\paperwidth]
					{palette tertiary}
					\begin{tabular}{ll}
						\multicolumn{2}{l}{\textcolor{white}{\textbf{\insertshortauthor}}}\\
						\multicolumn{2}{l}{\textcolor{white}{\insertinstitute}}\\
						\rule{0pt}{3ex}\\
						\if@german\textcolor{white}{E-Mail}\else\textcolor{white}{Mail}\fi & \textcolor{white}{\href{mailto:#1}{#1}}\\
						\if@german\textcolor{white}{Telefon}\else\textcolor{white}{Phone}\fi  & \textcolor{white}{#2}\\
						\if@german\textcolor{white}{Internet}\else\textcolor{white}{Internet}\fi & \textcolor{white}{\href{#3}{#3}}
					\end{tabular}
				\end{beamercolorbox}
			\end{column}
		\end{columns}	
	\end{frame}
}

\newlength{\circlesize}

\setbeamertemplate{section page}%
{
	\begin{tikzpicture}[overlay, remember picture]
	\setlength{\circlesize}{.9\textheight}
	\clip (current page.north west) rectangle (current page.south east);
	\shade[%
	top color=CD03,
	bottom color=CD02,
	shading angle=-45]
	(current page.north west) rectangle (current page.south east);
	\node[%
	circle,
	fill=white,
	anchor=north west,
	minimum size=\circlesize,
	text width=.85\circlesize,
	align=center,
	font=\bfseries\Huge]
	(sectiontitle) at ([xshift=.1\circlesize, yshift=-.2\circlesize]current page.north west) {\insertsection};
	\setlength{\circlesize}{.2\textheight}
	\node[%
	circle,
	fill=CD01,
	text=white,
	anchor=center,
	minimum size=\circlesize,
	align=left,
	font=\bfseries\Huge]
	at ([xshift=-.1\circlesize, yshift=-.3\circlesize]sectiontitle.south east) {\thesection};
	\end{tikzpicture}
}
